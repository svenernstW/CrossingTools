Getting Started with CrossingTools

library(CrossingTools)

n_ind   <- 20
n_mark  <- 200
n_trait <- 3

# Diploid genotype matrix: 0/2 coding for AA/BB
M  <- matrix(sample(c(0, 2), n_ind * n_mark, replace = TRUE), nrow = n_ind)

# Additive marker effects (markers in rows, traits in columns)
U  <- matrix(rnorm(n_mark * n_trait), ncol = n_trait)

# Simple 2-chromosome map, evenly spaced
map <- data.frame(
  site = 1:n_mark,
  chr  = rep(1:2, each = n_mark/2),
  pos  = c(seq(0, 100, length.out = n_mark/2),
           seq(0, 100, length.out = n_mark/2))
)

# Create pairwise crosses (scales as O(n^2))
crosses <- t(combn(n_ind, 2))

Doubled haploids (DH)

# We compute for doubled haploid (DH) families.
# method can be "osthushenrich" or "lehermeier".
# With multiple traits, setting covariance = TRUE returns trait covariances for segregation.
out_dh <- calculate_variances_DH(
  crosses         = crosses,   # (n_crosses x 2)
  genetic.map     = map,       # must contain site/chr/pos and cover all M columns
  M               = M,         # 0/2 genotype coding
  U               = U,         # additive marker effects (markers x traits)
  t               = 2,         # random-mating generations before DH creation
  intensity       = 1,         # selection differential for SPV
  covariance      = TRUE,      # also return segregation covariance matrices
  gains           = rep(10, n_trait),
  calculate.gains = TRUE,
  method          = "osthushenrich"
)

Heterozygous parents and F1 (additive + dominance)

# For heterozygous parents you provide haplotypes for each individual:
# hap1 and hap2 use 0/1 coding and have the same number of rows (individuals) and columns (markers).
hap1 <- matrix(sample(0:1, n_ind * n_mark, replace = TRUE), nrow = n_ind, ncol = n_mark)
hap2 <- matrix(sample(0:1, n_ind * n_mark, replace = TRUE), nrow = n_ind, ncol = n_mark)

# Dominance marker effects (markers in rows, traits in columns)
D <- matrix(rnorm(n_mark * n_trait), ncol = n_trait)

out_f1 <- calculate_variances_F1(
  crosses         = crosses,   # same cross list
  genetic.map     = map,
  hap1            = hap1,      # 0/1 haplotypes
  hap2            = hap2,      # 0/1 haplotypes
  U               = U,         # additive effects
  D               = D,         # dominance effects
  intensity       = 1,
  calculate.gains = TRUE,
  gains           = c(1, 1, 1),
  covariance      = TRUE
)
