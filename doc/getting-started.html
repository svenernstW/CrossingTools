Getting Started with CrossingTools

library(CrossingTools)

# ------------------------------------------------------------
# Basic Example Setup
# ------------------------------------------------------------
# In this section we create simple simulated data that will be used
# throughout the vignette to illustrate the functionality of CrossingTools.

n_ind   <- 20   # number of individuals
n_mark  <- 200  # number of markers
n_trait <- 3    # number of traits

# Generate a diploid genotype matrix (0 = AA, 2 = BB)
M <- matrix(sample(c(0, 2), n_ind * n_mark, replace = TRUE), nrow = n_ind)

# Simulate additive marker effects for all traits (markers x traits)
U <- matrix(rnorm(n_mark * n_trait), ncol = n_trait)

# Define a simple two-chromosome map with evenly spaced markersin Morgan
map <- data.frame(
  site = 1:n_mark,
  chr  = rep(1:2, each = n_mark / 2),
  pos  = c(seq(0, 1.5, length.out = n_mark / 2),
           seq(0, 1.5, length.out = n_mark / 2))
)

# Generate all possible pairwise crosses among the individuals.
# This scales as O(n^2) and can become large in realistic populations.
crosses <- t(combn(n_ind, 2))

Doubled Haploids (DH)

# ------------------------------------------------------------
# Doubled Haploid (DH) Families
# ------------------------------------------------------------
# The DH population type assumes complete homozygosity.
# The CrossingTools function `calculate_variances_DH` 
# estimates the expected genetic variance, superior progeny value,
# and optionally covariance among traits
# for all possible crosses.

# The method argument supports both "osthushenrich" and "lehermeier" for the 
# respective methods to calculate segregation variance.When multiple traits are
# included and covariance = TRUE, trait covariances are also computed.


out_dh <- calculate_variances_DH(
  crosses         = crosses,
  genetic.map     = map,       # must contain site, chr, and pos covering all M columns
  M               = M,         # genotype matrix (0/2 coding)
  U               = U,         # additive marker effects (markers x traits)
  t               = 2,         # random-mating generations before DH creation
  intensity       = 1,         # selection differential for SPV
  covariance      = TRUE,      # include segregation covariance matrices
  method          = "osthushenrich"
)
#> [1] "length(gains)!= ncol(U), ignoring desired gains related arguments"

Clonal Breeding with Heterozygous Parents and F1s (Additive + Dominance)

# ------------------------------------------------------------
# Clonal breeding: Heterozygous Parents (Additive + Dominance Effects)
# ------------------------------------------------------------
# For clonal breeding When parents are heterozygous, both additive and dominance effects
# contribute to offspring performance. To model this, we provide haplotype
# matrices for each parent: hap1 and hap2 (0/1 coding).

hap1 <- matrix(sample(0:1, n_ind * n_mark, replace = TRUE), nrow = n_ind, ncol = n_mark)
hap2 <- matrix(sample(0:1, n_ind * n_mark, replace = TRUE), nrow = n_ind, ncol = n_mark)

# Simulate dominance marker effects (markers x traits)
D <- matrix(rnorm(n_mark * n_trait), ncol = n_trait)

out_f1 <- calculate_variances_F1(
  crosses         = crosses,
  genetic.map     = map,
  hap1            = hap1,
  hap2            = hap2,
  U               = U,
  D               = D,
  intensity       = 1,
  calculate.gains = TRUE,
  gains           = c(1, 1, 1),
  covariance      = TRUE
)

Optimal Cross Selection (OCS)

# ------------------------------------------------------------
# Optimal Cross Selection (OCS)
# ------------------------------------------------------------
# OCS identifies an optimal subset of crosses balancing expected genetic gain
# and diversity using a genetic algorithm (GA). It can use all values calculated above
# such as EGBV or SPV

# Compute a genomic relationship matrix (G) as a measure of relatedness.
G <- tcrossprod(scale(M, center = TRUE, scale = FALSE)) / n_mark

# Select a cross-level criterion, here SPV for the first trait.
crit <- out_dh$cross_values$SPV1

# Execute OCS optimization.
ocs <- optimal_cross_selection(
  crosses        = crosses,
  ncrosses       = 20,
  G              = G,
  u              = crit,
  target.angle   = 30 * pi / 180,  # trade-off between gain and diversity (radians)
  n.select       = 200,            # GA parent pool size
  n.pop          = 1000,           # GA population size
  max.generation = 200,            # max GA iterations
  max.iteration  = 50,             # early stopping threshold
  angle.penalty  = 0.01,           # penalty for angle deviation
  n.Threads      = 2L
)

Multi-Trait Optimization with CrossingTools

# ------------------------------------------------------------
# Multi-Trait Optimization and Desired Gains Index (DGI)
# ------------------------------------------------------------
# Multi-trait optimization in CrossingTools is based on the Desired Gains Index (DGI).
# The DGI provides a unified framework to achieve user-specified improvement goals across
# multiple traits, by combining expected means and covariances into a single index.

# Example scenario:
# We aim to increase the first two traits equally while keeping the third constant.
gains <- c(1, 1, 0)

# ------------------------------------------------------------
# Approach 1: DGI based on Genomic BLUPs (A) and posterior variance (V)
# ------------------------------------------------------------
# Following Werner et al. (2024), this method uses genomic BLUPs (A) as predicted
# genetic values per individual and their associated (co)variance structure (V).
# When posterior variances are unavailable, the covariance matrix of A can be used
# as an approximation (V.approx). This assumes balanced data.

A <- M %*% U         # predicted genetic values per trait
V.approx <- cov(A)   # approximate posterior covariance of BLUPs

IDG <- calculate_desired_gains(
  A             = A,
  V.approx      = V.approx,
  gains         = gains,
  use.V.approx  = TRUE,
  use.V = FALSE,
  use.marginal.V = F
)

# Returns a single-column data.frame containing DGI scores per individual.
head(IDG)
#>      DG_index
#> 1  0.01459255
#> 2 -0.15084393
#> 3 -0.24681194
#> 4 -0.23975509
#> 5 -0.27036839
#> 6 -0.30914515

# ------------------------------------------------------------
# Approach 2: DGI based on Segregation (Co)variances from Crosses
# ------------------------------------------------------------
# This approach estimates the DGI at the cross level by leveraging segregation
# variances. This can be interpreted as a posterior segregation variance based
# on marker BLUPs.
out_dh <- calculate_variances_DH(
  crosses         = crosses,
  genetic.map     = map,
  M               = M,
  U               = U,
  t               = 2,
  intensity       = 1,
  covariance      = TRUE,
  gains           = gains,
  calculate.gains = TRUE,
  method          = "osthushenrich"
)

# The resulting $gains component summarizes expected multi-trait progress
# under the specified desired gains vector.
